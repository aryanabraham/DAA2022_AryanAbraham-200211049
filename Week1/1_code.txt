#include <iostream>
using namespace std;

int main ()
{
	int i;
	int n;
	int t;
	int key;
	bool flag;

	cin>>t;
	while(t--)
	{
		flag=true;
		cin>>n;
		int a[n];
		
		for(i=0;i<n;i++)
			cin>>a[i];

		cin>>key;

		for(i=0;i<n;i++)
		{
			if(a[i]==key)
			{
				cout<<"Present "<<i+1<<endl;
				flag=false;
				break;
			}
		}

		if(flag)
			cout<<"Not Present "<<n<<endl;
	}
	return 0;
}
